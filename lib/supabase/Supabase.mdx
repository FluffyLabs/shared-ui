# Supabase Integration

The `@fluffylabs/shared-ui/supabase` subpath provides authentication, user menu, settings, and per-user data storage components powered by [Supabase](https://supabase.com).

All Fluffy Labs apps share a single Supabase project, so users have one account across all apps.

## Prerequisites

Install the Supabase JS SDK alongside the shared UI:

```bash
npm install @supabase/supabase-js
```

## Supabase Project Setup

### 1. Create the `user_data` table

Run this SQL in your Supabase SQL editor (Dashboard > SQL Editor):

```sql
create table user_data (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  app_id text not null default '',
  key text not null,
  value jsonb,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique (user_id, app_id, key)
);

-- Row Level Security: users can only access their own data
alter table user_data enable row level security;

create policy "Users can manage their own data"
  on user_data for all
  using (auth.uid() = user_id)
  with check (auth.uid() = user_id);
```

### 2. Create the `subscriptions` and `usage` tables

```sql
-- Subscription status per user (managed by Stripe webhooks via Edge Function)
create table subscriptions (
  user_id uuid primary key references auth.users(id) on delete cascade,
  status text not null default 'free',
  stripe_customer_id text,
  stripe_subscription_id text,
  current_period_end timestamptz,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

alter table subscriptions enable row level security;

create policy "Users can read their own subscription"
  on subscriptions for select
  using (auth.uid() = user_id);

-- Usage tracking for rate limiting (free plan)
create table usage (
  user_id uuid not null references auth.users(id) on delete cascade,
  app_id text not null default '',
  action text not null,
  period text not null,
  count int not null default 0,
  primary key (user_id, app_id, action, period)
);

alter table usage enable row level security;

create policy "Users can read their own usage"
  on usage for select
  using (auth.uid() = user_id);

-- Atomic increment function (called by useQuota)
create or replace function increment_usage(
  p_user_id uuid,
  p_app_id text,
  p_action text,
  p_period text
) returns int as $$
declare
  new_count int;
begin
  -- Ensure caller can only increment their own usage
  if auth.uid() is null or p_user_id is distinct from auth.uid() then
    raise exception 'Access denied: cannot modify another user''s usage';
  end if;

  insert into usage (user_id, app_id, action, period, count)
  values (p_user_id, p_app_id, p_action, p_period, 1)
  on conflict (user_id, app_id, action, period)
  do update set count = usage.count + 1
  returning count into new_count;
  return new_count;
end;
$$ language plpgsql security definer;
```

### 3. Configure Auth

In the Supabase dashboard under **Authentication > Settings**:

- Set the **Site URL** to your production domain
- Add **Redirect URLs** for your magic link callback route (e.g., `http://localhost:5173/auth/callback`, `https://your-app.com/auth/callback`)

## Usage

### Wrap your app with `SupabaseProvider`

```tsx
import { SupabaseProvider } from "@fluffylabs/shared-ui/supabase";

function App() {
  return (
    <SupabaseProvider
      supabaseUrl="https://your-project.supabase.co"
      supabaseAnonKey="your-anon-key"
      appId="my-app"
    >
      {/* your app */}
    </SupabaseProvider>
  );
}
```

The `appId` is used by `useUserData` to distinguish per-app data from shared data.

### Add UserMenu to the Header

```tsx
import { Header } from "@fluffylabs/shared-ui/components";
import { UserMenu } from "@fluffylabs/shared-ui/supabase";

<Header
  toolNameSrc={toolNameLogo}
  endSlot={
    <UserMenu
      onLoginClick={() => navigate("/login")}
      onSettingsClick={() => navigate("/settings")}
    />
  }
/>
```

### Add the AuthFlow page

The auth flow is magic-link-first: users enter their email, receive a link, and click it to sign in. Password login is available as a secondary option for users who have set a password in Settings.

```tsx
import { AuthFlow } from "@fluffylabs/shared-ui/supabase";

function LoginPage() {
  return (
    <AuthFlow
      onSuccess={() => navigate("/")}
      redirectTo={window.location.origin + "/auth/callback"}
    />
  );
}
```

The `redirectTo` URL is where Supabase sends users after they click the magic link. You must add this URL to your Supabase project's **Redirect URLs** allow-list.

### Add the AuthCallback page

Mount `AuthCallback` at the route matching your `redirectTo` URL. It handles the token exchange and redirects on success.

```tsx
import { AuthCallback } from "@fluffylabs/shared-ui/supabase";

function AuthCallbackPage() {
  return (
    <AuthCallback
      onSuccess={() => navigate("/")}
      onError={(err) => console.error(err)}
    />
  );
}
```

### Add the Settings page

Settings includes theme selection and password management. Users who signed in via magic link can set a password here for convenience.

```tsx
import { Settings } from "@fluffylabs/shared-ui/supabase";

function SettingsPage() {
  return <Settings />;
}
```

### Store per-user data

```tsx
import { useUserData } from "@fluffylabs/shared-ui/supabase";

function MyComponent() {
  // Shared across all apps (app_id = null)
  const { data: theme, save: saveTheme } = useUserData<string>("theme");

  // Scoped to this app only (uses appId from SupabaseProvider)
  const { data: bookmarks, save: saveBookmarks } = useUserData<string[]>(
    "bookmarks",
    { appScoped: true }
  );
}
```

### Gate premium features with quota

```tsx
import { QuotaGate } from "@fluffylabs/shared-ui/supabase";

function DecoderPage() {
  return (
    <QuotaGate action="decode" freeLimit={50} onCheckout={handleCheckout}>
      <Decoder />
    </QuotaGate>
  );
}
```

Free users see a usage progress bar and can use the feature up to 50 times/month.
When the limit is reached, a pricing card is shown instead.
Pro users always see the content with no limit.

### Show subscription status in Settings

```tsx
import { Settings, SubscriptionStatus } from "@fluffylabs/shared-ui/supabase";

function SettingsPage() {
  return (
    <>
      <Settings />
      <SubscriptionStatus
        onCheckout={() => handleCheckout()}
        onManage={() => openStripePortal()}
      />
    </>
  );
}
```

### Use quota hook directly

```tsx
import { useQuota } from "@fluffylabs/shared-ui/supabase";

function DecoderTool() {
  const { canUse, remaining, increment } = useQuota("decode", { freeLimit: 50 });

  async function handleDecode() {
    if (!canUse) return; // show upgrade prompt
    await increment();
    // perform the action
  }
}
```

### Access auth state

```tsx
import { useUser, useSession, useSignOut } from "@fluffylabs/shared-ui/supabase";

function MyComponent() {
  const { user, isLoading } = useUser();
  const { session, isLoading: sessionLoading } = useSession();
  const signOut = useSignOut();

  if (isLoading) return <span>Loading...</span>;
  if (!user) return <span>Not logged in</span>;
  return <button onClick={signOut}>Sign out {user.email}</button>;
}
```

## Components & Hooks

| Export | Description |
|--------|-------------|
| `SupabaseProvider` | Context provider — wraps your app, creates the Supabase client |
| `AuthFlow` | Magic-link-first auth screen with optional password login |
| `AuthCallback` | Handles magic link redirect — mount at your callback route |
| `PasswordUpdate` | Set/change password form (embedded in Settings) |
| `UserMenu` | Header dropdown — login button (logged out) or email + settings + sign out (logged in) |
| `Settings` | Settings panel with theme selector and password management |
| `useUser` | Returns `{ user, isLoading }` |
| `useSession` | Returns `{ session, isLoading }` |
| `useSignOut` | Returns a sign-out callback |
| `useUserData` | Per-user key-value storage with `{ appScoped }` option |
| `useSubscription` | Returns `{ subscription, isLoading }` with plan and status |
| `useQuota` | Rate-limited usage tracking with `{ freeLimit }` option |
| `PricingCard` | Pro plan card with price, features, and checkout button |
| `QuotaGate` | Gates content behind subscription + free quota |
| `SubscriptionStatus` | Plan badge with manage/upgrade button (for Settings) |

## Stripe Edge Functions

The `supabase/functions/` directory contains three Edge Functions that handle Stripe operations
requiring a server-side secret key. These are deployed to your Supabase project, not published
as part of the npm package.

### Setup

1. **Create a Stripe account** and get your API keys from the [Stripe Dashboard](https://dashboard.stripe.com/apikeys)

2. **Create a Product and Price** in Stripe for the monthly Pro plan. Copy the Price ID (starts with `price_`).

3. **Set environment variables** in your Supabase project (Dashboard > Edge Functions > Secrets):

   ```
   STRIPE_SECRET_KEY=sk_live_...
   STRIPE_WEBHOOK_SECRET=whsec_...
   STRIPE_PRICE_ID=price_...
   SITE_URL=https://your-app.com
   ```

4. **Deploy the Edge Functions**:

   ```bash
   supabase functions deploy create-checkout
   supabase functions deploy stripe-webhook
   supabase functions deploy create-portal
   ```

5. **Create a Stripe webhook** pointing to your `stripe-webhook` function URL:
   `https://<project-ref>.supabase.co/functions/v1/stripe-webhook`

   Subscribe to these events:
   - `checkout.session.completed`
   - `customer.subscription.updated`
   - `customer.subscription.deleted`

6. **Configure the Customer Portal** in [Stripe Dashboard > Settings > Customer Portal](https://dashboard.stripe.com/settings/billing/portal)
   to allow customers to cancel their subscription.

### Calling from your app

```tsx
import { useSession } from "@fluffylabs/shared-ui/supabase";

const SUPABASE_URL = "https://your-project.supabase.co";

function useCheckout() {
  const { session } = useSession();

  return async () => {
    const res = await fetch(`${SUPABASE_URL}/functions/v1/create-checkout`, {
      method: "POST",
      headers: {
        Authorization: `Bearer ${session?.access_token}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ returnUrl: window.location.origin }),
    });
    const { url } = await res.json();
    window.location.href = url;
  };
}

function useManageSubscription() {
  const { session } = useSession();

  return async () => {
    const res = await fetch(`${SUPABASE_URL}/functions/v1/create-portal`, {
      method: "POST",
      headers: {
        Authorization: `Bearer ${session?.access_token}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ returnUrl: window.location.origin }),
    });
    const { url } = await res.json();
    window.location.href = url;
  };
}
```

Then wire them into your components:

```tsx
function SettingsPage() {
  const checkout = useCheckout();
  const manage = useManageSubscription();

  return (
    <>
      <Settings />
      <SubscriptionStatus onCheckout={checkout} onManage={manage} />
    </>
  );
}
```
